def word(teste):
    sleep(2)

    if(exists()):
        click()
    else:
        find()
        click()
        type('word' + Key.ENTER)
        sleep(2)
        wait()
        type(Key.ENTER)
    type('setor '+ teste + Key.ENTER)
    type('v', KEY_CTRL)
    find()
    click()
