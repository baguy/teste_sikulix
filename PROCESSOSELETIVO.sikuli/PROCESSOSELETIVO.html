    sleep(2)
    type(Key.PRINTSCREEN)
    if(exists()):
        click()
    else:
        find()
        click()
        type('word' + Key.ENTER)
        sleep(2)
        wait()
        type(Key.ENTER)
    type('processo seletivo '+ teste + Key.ENTER)
    type('v', KEY_CTRL)
    find()
